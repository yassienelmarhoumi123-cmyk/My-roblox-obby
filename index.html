import pygame import sys import random # Initialisierung pygame.init() # Konstanten SCREEN_WIDTH = 400 SCREEN_HEIGHT = 600 GRAVITY = 0.25 FLAP_STRENGTH = -6 PIPE_SPEED = 3 PIPE_GAP = 150 # Farben WHITE = (255, 255, 255) SKY_BLUE = (135, 206, 235) BIRD_YELLOW = (255, 255, 0) PIPE_GREEN = (34, 139, 34) screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) clock = pygame.time.Clock() font = pygame.font.SysFont("Arial", 32) def draw_text(text, x, y): img = font.render(text, True, WHITE) screen.blit(img, (x, y)) def main(): bird_rect = pygame.Rect(50, SCREEN_HEIGHT // 2, 30, 30) bird_movement = 0 pipes = [] spawn_pipe_event = pygame.USEREVENT pygame.time.set_timer(spawn_pipe_event, 1200) score = 0 game_active = True while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: if game_active: bird_movement = FLAP_STRENGTH else: # Neustart main() if event.type == spawn_pipe_event and game_active: pipe_height = random.randint(150, 400) top_pipe = pygame.Rect(SCREEN_WIDTH, 0, 50, pipe_height - PIPE_GAP // 2) bottom_pipe = pygame.Rect(SCREEN_WIDTH, pipe_height + PIPE_GAP // 2, 50, SCREEN_HEIGHT) pipes.append(top_pipe) pipes.append(bottom_pipe) screen.fill(SKY_BLUE) if game_active: # Vogel Bewegung bird_movement += GRAVITY bird_rect.centery += int(bird_movement) pygame.draw.ellipse(screen, BIRD_YELLOW, bird_rect) # Röhren Bewegung for pipe in pipes: pipe.centerx -= PIPE_SPEED pygame.draw.rect(screen, PIPE_GREEN, pipe) # Röhren entfernen, wenn sie aus dem Bild sind pipes = [pipe for pipe in pipes if pipe.right > 0] # Kollisionen for pipe in pipes: if bird_rect.colliderect(pipe): game_active = False if bird_rect.top <= 0 or bird_rect.bottom >= SCREEN_HEIGHT: game_active = False # Score (vereinfacht: zählt passierte Röhren-Paare) score += 0.01 draw_text(f"Score: {int(score)}", 10, 10) else: draw_text("GAME OVER", 120, 250) draw_text("Drücke LEERTASTE für Neustart", 30, 300) pygame.display.update() clock.tick(60) if __name__ == "__main__": main()
